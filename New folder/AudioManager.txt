import { SynthEngine } from './SynthEngine';
import { EngineSound } from './EngineSound';
import { EngineNode } from './EngineNode';

/**
 * AudioManager — combines AudioWorklet (realistic) + SynthEngine (fallback) + EngineSound (samples).
 */
export class AudioManager {
  private ctx: AudioContext | null = null;
  private masterGain: GainNode | null = null;

  // Primary: AudioWorklet (per-cylinder combustion simulation)
  private engineNode: EngineNode | null = null;

  // Fallback: Oscillator-based synthesis
  private synthEngine: SynthEngine | null = null;

  // Sample-based sound
  private engineSound: EngineSound | null = null;

  private volume = 0.7;
  private initialized = false;
  private engineRunning = false;
  private useWorklet = false;

  async init(): Promise<void> {
    if (this.initialized) return;

    try {
      this.ctx = new AudioContext();
      this.masterGain = this.ctx.createGain();
      this.masterGain.gain.value = 0;
      this.masterGain.connect(this.ctx.destination);

      // Init AudioWorklet (primary)
      this.engineNode = new EngineNode(this.ctx, this.masterGain);
      await this.engineNode.init();

      if (this.engineNode.isReady()) {
        this.useWorklet = true;
        console.log('[AudioManager] Menggunakan AudioWorklet (realistic)');
      } else {
        // Fallback ke SynthEngine
        this.synthEngine = new SynthEngine(this.ctx, this.masterGain);
        this.synthEngine.start();
        console.log('[AudioManager] Fallback ke SynthEngine (oscillator)');
      }

      // Sample-based engine sound
      this.engineSound = new EngineSound(this.ctx, this.masterGain);

      this.initialized = true;
      console.log('[AudioManager] Berhasil di-init');
    } catch (error) {
      console.error('[AudioManager] Gagal init:', error);
    }
  }

  setEngineConfig(cylinders: number, redline: number, engineType?: string): void {
    // Map engine ID ke tipe AudioWorklet
    const typeMap: Record<string, string> = {
      'lexus-lfa-v10': 'V10',
      'subaru-ej25': 'BOXER_4',
      'toyota-2jz': 'INLINE_6',
      'honda-f20c': 'INLINE_4',
      'bmw-m52b28': 'INLINE_6',
      'ferrari-f136-v8': 'V8_CROSSPLANE',
    };

    const workletType = engineType || typeMap[this.currentConfigId] || 'V10';

    this.engineNode?.setConfig(workletType, cylinders);
    this.synthEngine?.setEngineConfig(cylinders, redline);
  }

  private currentConfigId = '';

  async loadEngineSounds(
    configId: string,
    samples: Record<string, string>
  ): Promise<void> {
    if (!this.ctx || !this.masterGain) {
      await this.init();
    }
    if (!this.ctx || !this.masterGain) return;

    this.currentConfigId = configId;

    try {
      if (this.engineSound) {
        await this.engineSound.loadSamples(samples);
      }
    } catch (error) {
      console.error(`[AudioManager] Gagal load sound ${configId}:`, error);
    }
  }

  update(
    rpm: number,
    throttle: number,
    atRevLimiter: boolean,
    crossfadeRPM: number,
    isRunning: boolean
  ): void {
    if (!this.initialized || !this.masterGain || !this.ctx) return;

    // Master gate: mute saat engine mati
    if (isRunning !== this.engineRunning) {
      this.engineRunning = isRunning;
      this.masterGain.gain.setTargetAtTime(
        isRunning ? this.volume : 0,
        this.ctx.currentTime,
        0.05
      );
    }

    // Update AudioWorklet (primary)
    if (this.useWorklet && this.engineNode) {
      this.engineNode.update(rpm, throttle);
    }

    // Update SynthEngine (fallback)
    if (!this.useWorklet && this.synthEngine) {
      this.synthEngine.update(rpm, throttle, atRevLimiter);
    }

    // Update sample-based sound
    this.engineSound?.update(rpm, throttle, crossfadeRPM);
  }

  setVolume(v: number): void {
    this.volume = Math.max(0, Math.min(1, v));
    if (this.masterGain && this.ctx && this.engineRunning) {
      this.masterGain.gain.setTargetAtTime(this.volume, this.ctx.currentTime, 0.01);
    }
  }

  resume(): void {
    if (this.ctx?.state === 'suspended') {
      this.ctx.resume();
    }
  }

  destroy(): void {
    this.engineNode?.disconnect();
    this.synthEngine?.stop();
    this.engineSound?.stopAll();
    this.ctx?.close();
    this.ctx = null;
    this.initialized = false;
  }
}
