/**
 * EngineProcessor — AudioWorklet processor untuk suara mesin realistis.
 * Jalan di audio thread (bukan main thread).
 *
 * Fitur:
 * - Per-cylinder combustion pulse simulation
 * - Firing order & offset per tipe mesin (I4, Boxer, I6, V8, V10)
 * - Soft clipping (tanh) — natural overdrive
 * - Valve train mechanical noise
 * - RPM-dependent filtering
 */

// @ts-nocheck — AudioWorklet scope punya globals yang gak dikenal TypeScript

declare const sampleRate: number;
declare const registerProcessor: (name: string, ctor: typeof AudioWorkletProcessor) => void;

class EngineProcessor extends AudioWorkletProcessor {
  phase = 0;
  numCylinders = 10;
  firingOffsets: number[] = [];
  engineType = 'V10';

  static get parameterDescriptors() {
    return [
      { name: 'rpm', defaultValue: 0, minValue: 0, maxValue: 12000 },
      { name: 'throttle', defaultValue: 0, minValue: 0, maxValue: 1 },
    ];
  }

  constructor() {
    super();
    this.port.onmessage = (event) => {
      if (event.data.type === 'config') {
        this.engineType = event.data.engineType || 'V10';
        this.numCylinders = event.data.numCylinders || 10;
        this.firingOffsets = this.calculateFiringOffsets();
      }
    };
    this.firingOffsets = this.calculateFiringOffsets();
  }

  /**
   * Hitung firing offset berdasarkan tipe mesin.
   * Setiap tipe punya karakteristik suara beda.
   */
  calculateFiringOffsets(): number[] {
    const n = this.numCylinders;
    const pi2 = Math.PI * 2;
    const pi4 = Math.PI / 2;

    switch (this.engineType) {
      case 'INLINE_4': {
        // Honda F20C — fire tiap 180°, uneven untuk VTEC character
        return [0, pi4 * 2, pi4, pi4 * 3];
      }
      case 'BOXER_4': {
        // Subaru EJ25 — unequal length headers, ciri khas "rumble"
        return [0, pi4 * 2.2, pi4 * 0.8, pi4 * 3.1];
      }
      case 'INLINE_6': {
        // Toyota 2JZ / BMW M52 — firing 120°, halus & smooth
        return [0, pi2 / 3, (pi2 / 3) * 2, pi2 / 3, (pi2 / 3) * 2, pi2 / 3];
      }
      case 'V8_CROSSPLANE': {
        // Ferrari F136 — crossplane firing uneven
        return [0, pi4, pi4 * 2, pi4 * 3, pi4 * 0.5, pi4 * 1.5, pi4 * 2.5, pi4 * 3.5];
      }
      case 'V10': {
        // Lexus LFA — V10 72° firing, harmonik tinggi
        const offsets = [];
        for (let i = 0; i < 10; i++) {
          offsets.push((i * pi2) / 10);
        }
        return offsets;
      }
      default: {
        const offsets = [];
        for (let i = 0; i < n; i++) {
          offsets.push((i * pi2) / n);
        }
        return offsets;
      }
    }
  }

  /**
   * Proses audio sample-by-sample.
   * Harus selesai dalam ~2.9ms (128 samples @ 44.1kHz).
   */
  process(inputs, outputs, parameters) {
    const output = outputs[0];
    if (!output || !output[0]) return true;

    const channel = output[0];
    const numSamples = channel.length;

    // Ambil parameter RPM dan throttle
    const rpmArr = parameters.rpm;
    const throttleArr = parameters.throttle;

    for (let i = 0; i < numSamples; i++) {
      const currentRPM = rpmArr.length > 1 ? rpmArr[i] : rpmArr[0];
      const currentThrottle = throttleArr.length > 1 ? throttleArr[i] : throttleArr[0];

      // ═══ 1. FIRING FREQUENCY ═══
      // f₀ = RPM/60 × cylinders/2 (4-stroke)
      const firingFreq = (currentRPM / 60) * (this.numCylinders / 2);

      // ═══ 2. COMBUSTION PULSE ═══
      let totalPressure = 0;
      const rpmNorm = Math.min(1, currentRPM / 9000);

      for (let c = 0; c < this.numCylinders; c++) {
        const cylinderOffset = this.firingOffsets[c] || 0;
        const cylinderPhase = (this.phase + cylinderOffset) % (Math.PI * 2);

        // Combustion pulse — exponential curve saat compression
        if (cylinderPhase < Math.PI) {
          const t = cylinderPhase / Math.PI;
          const combustionCurve = Math.sin(cylinderPhase) * Math.exp(-t * 3);
          const intensity = 0.3 + currentThrottle * 0.7;
          totalPressure += combustionCurve * intensity;
        }
      }

      // Normalisasi
      totalPressure /= Math.max(1, this.numCylinders / 4);

      // ═══ 3. MECHANICAL NOISE (Valve Train) ═══
      let valveNoise = 0;
      const mechPhase = Math.sin(this.phase * this.numCylinders * 2);
      if (mechPhase > 0.85) {
        valveNoise = (Math.random() - 0.5) * 0.3 * rpmNorm;
      }

      // ═══ 4. KOMBINASI ═══
      let sample = totalPressure + valveNoise;

      // ═══ 5. SOFT CLIPPING (tanh) ═══
      // Natural overdrive — suara "hangat" kayak knalpot asli
      sample = Math.tanh(sample * 2.5);

      // ═══ 6. RPM-DEPENDENT LEVEL ═══
      sample *= 0.7 + rpmNorm * 0.3;

      // Tulis output
      channel[i] = sample;
      if (output.length > 1) {
        output[1][i] = sample * 0.95;
      }
    }

    // Update phase
    const lastRPM = rpmArr.length > 1 ? rpmArr[numSamples - 1] : rpmArr[0];
    const lastFiringFreq = (lastRPM / 60) * (this.numCylinders / 2);
    this.phase += lastFiringFreq * 2 * Math.PI * numSamples / sampleRate;
    this.phase %= Math.PI * 2 * 100;

    return true;
  }
}

registerProcessor('engine-processor', EngineProcessor);
