/**
 * EngineNode — Main thread wrapper untuk EngineProcessor AudioWorklet.
 * Handle init, parameter updates, dan routing audio.
 */
export class EngineNode {
  private ctx: AudioContext;
  private workletNode: AudioWorkletNode | null = null;
  private output: GainNode;
  private initialized = false;

  constructor(ctx: AudioContext, output: GainNode) {
    this.ctx = ctx;
    this.output = output;
  }

  /** Init AudioWorklet — harus dipanggil setelah user gesture */
  async init(): Promise<void> {
    if (this.initialized) return;

    try {
      // Register AudioWorklet processor
      // Path harus relative ke root, bukan ke file
      await this.ctx.audioWorklet.addModule('/src/audio/worklets/engine-processor.ts');

      // Buat worklet node
      this.workletNode = new AudioWorkletNode(this.ctx, 'engine-processor', {
        numberOfInputs: 0,
        numberOfOutputs: 1,
        outputChannelCount: [2], // Stereo
      });

      // Route: worklet → output
      this.workletNode.connect(this.output);

      this.initialized = true;
      console.log('[EngineNode] AudioWorklet initialized');
    } catch (error) {
      console.error('[EngineNode] Gagal init AudioWorklet:', error);
      console.log('[EngineNode] Fallback ke SynthEngine');
    }
  }

  /** Update engine config (tipe mesin, jumlah silinder) */
  setConfig(engineType: string, numCylinders: number): void {
    this.workletNode?.port.postMessage({
      type: 'config',
      engineType,
      numCylinders,
    });
  }

  /** Update RPM dan throttle — setiap frame */
  update(rpm: number, throttle: number): void {
    if (!this.workletNode) return;

    // Set parameter values via AudioParam
    const rpmParam = this.workletNode.parameters.get('rpm');
    const throttleParam = this.workletNode.parameters.get('throttle');

    if (rpmParam) rpmParam.setValueAtTime(rpm, this.ctx.currentTime);
    if (throttleParam) throttleParam.setValueAtTime(throttle, this.ctx.currentTime);
  }

  isReady(): boolean {
    return this.initialized;
  }

  disconnect(): void {
    this.workletNode?.disconnect();
    this.workletNode = null;
    this.initialized = false;
  }
}
